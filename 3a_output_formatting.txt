import requests
import json

def emotion_detector(text_to_analyze: str):
    response = requests.post(url='https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict',
    headers = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"},
    json = { "raw_document": { "text": text_to_analyze } })

    jsonObject = json.loads(response.text)
    anger_score = jsonObject['emotionPredictions'][0]['emotion']['anger']
    disgust_score = jsonObject['emotionPredictions'][0]['emotion']['disgust']
    fear_score = jsonObject['emotionPredictions'][0]['emotion']['fear']
    joy_score = jsonObject['emotionPredictions'][0]['emotion']['joy']
    sadness_score = jsonObject['emotionPredictions'][0]['emotion']['sadness']
    
    

    emotionDict = {
        'anger' : anger_score,
        'disgust' : disgust_score,
        'fear' : fear_score,
        'joy' : joy_score,
        'sadness' : sadness_score,        
    }

    dominant_emotion = get_dominant_emotion(emotionDict)
    emotionDict.update({'dominant_emotion' : dominant_emotion})

    return emotionDict;

def get_dominant_emotion(emotion_scores):
    maxkey = ""
    maxValue = 0
    
    for k in emotion_scores.keys():        
        v = emotion_scores.get(k)
        
        if(v > maxValue):
            maxkey = k
            maxValue = v

    return maxkey
